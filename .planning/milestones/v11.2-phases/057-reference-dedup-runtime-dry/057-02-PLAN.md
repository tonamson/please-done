```javascript
'use strict';

const fs = require('fs');
const path = require('path');

/**
 * Tạo thư mục recursive. Wrapper cho fs.mkdirSync({ recursive: true }).
 * @param {string} dirPath - Đường dẫn thư mục cần tạo
 */
function ensureDir(dirPath) {
  fs.mkdirSync(dirPath, { recursive: true });
}

/**
 * Kiểm tra .git tồn tại tại working directory.
 * Throw Error nếu không phải git repo.
 * @param {string} [dir] - Thư mục kiểm tra, mặc định process.cwd()
 */
function validateGitRoot(dir) {
  if (!fs.existsSync(path.join(dir || process.cwd(), '.git'))) {
    throw new Error('Khong phai git repo — khong tim thay .git');
  }
}

/**
 * Copy file, tạo backup nếu dest đã tồn tại.
 * Backup file: dest + '.bak'
 * @param {string} src - File nguồn
 * @param {string} dest - File đích
 */
function copyWithBackup(src, dest) {
  if (fs.existsSync(dest)) {
    fs.copyFileSync(dest, dest + '.bak');
  }
  fs.copyFileSync(src, dest);
}

module.exports = { ensureDir, validateGitRoot, copyWithBackup };
```
