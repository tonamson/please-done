    ```bash
    # Test converter output size difference
    node -e "
    const { parseFrontmatter, inlineWorkflow } = require('./bin/lib/utils');
    const { encodingForModel } = require('js-tiktoken');
    const fs = require('fs');
    const path = require('path');
    const enc = encodingForModel('gpt-4o');
    const ROOT = process.cwd();
    const COMMANDS = path.join(ROOT, 'commands', 'pd');

    const skills = ['write-code', 'plan', 'new-milestone', 'complete-milestone', 'fix-bug', 'what-next'];
    let totalConditional = 0;

    for (const name of skills) {
      const content = fs.readFileSync(path.join(COMMANDS, name + '.md'), 'utf8');
      const { body } = parseFrontmatter(content);
      const inlined = inlineWorkflow(body, ROOT);
      const conditional = inlined.match(/<conditional_reading>([\s\S]*?)<\/conditional_reading>/);
      if (conditional) {
        const tokens = enc.encode(conditional[1]).length;
        console.log(name + ': ' + tokens + ' tokens in conditional_reading (skippable)');
        totalConditional += tokens;
      }
    }
    console.log('Total skippable tokens across skills: ' + totalConditional);
    enc.free();
    "
    ```
